mx-service-picker directive API:


            <mx-service-picker
                ng-disabled = "demoCtrl.isDisabled"
                required    = "demoCtrl.isRequired"

                label       = "Service picker"

                ng-model    = "demoCtrl.model"
            >

            
Binding settings:


      
         bindToController: {

            // Boolean
                  disabled : "=ngDisabled",
                  required : "=",

            // String
                  label    : "@",

            // Object
                  model    : "=ngModel"

         }
