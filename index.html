Required Disabled Read-only {{ mode }} 
            
mx-service-picker directive API:


            <mx-service-picker
                ng-disabled      = "demoCtrl.isDisabled"
                required         = "demoCtrl.isRequired"
                read-only        = "demoCtrl.isReadonly"

                label            = "Service picker"

                ng-model         = "demoCtrl.someModel"

                navigate-host    = "demoCtrl.onNavigateHost"
                navigate-service = "demoCtrl.onNavigateService"
            >

            
Binding settings:


      
         bindToController: {

            // Boolean
                  disabled          : "=ngDisabled",
                  required          : "=",
                  readOnly          : "="

            // String
                  label             : "@",

            // Function
                   navigateHost    : "=",
                   navigateService : "=",

            // Object
                  model             : "=ngModel"

         }
